Define add_d!(string n) (warp=true)
{
    Repeat (n)
    {
        List.Add(followx, (MAP X + Motion.XPosition()));
        List.Add(followy, (MAP Y + Motion.YPosition()));
        List.Add(followd, Motion.Direction());
    }
}

Define FiddleOut (warp=true)
{
    Looks.SetSizeTo(44);
    F = 1;
    Motion.ChangeYBy(-16);
    If (Not (Sensing.TouchingObject(Map)))
    {
        F = 0.01;
    }
    If (F == 1)
    {
        Repeat Until (F == 0.01)
        {
            Motion.Move(F);
            If (Not (Sensing.TouchingObject(Map)))
            {
                F = 0;
            }
            Motion.Move((0 - F));
            F += 0.01;
            Motion.TurnRight(22.5);
        }
        Motion.PointInDirection(Motion.XPosition());
        F = -1;
    }
    Motion.ChangeYBy(16);
    Looks.SetSizeTo(100);
}

Define Load (warp=true)
{
    List.ReplaceItem(ItemCounter, 1, save data[1]);
    List.ReplaceItem(ItemCounter, 2, save data[2]);
    List.ReplaceItem(ItemCounter, 3, save data[3]);
    List.ReplaceItem(ItemCounter, 4, save data[4]);
    List.ReplaceItem(ItemCounter, 5, save data[5]);
    List.ReplaceItem(ItemCounter, 6, save data[6]);
    List.ReplaceItem(Character exp, 1, save data[7]);
    List.ReplaceItem(Character exp, 2, save data[8]);
    List.ReplaceItem(Character exp, 3, save data[9]);
    AREA = save data[10];
    RBtoggle = save data[12];
    If (save data[11] == "true")
    {
        MAP X = save2x[save data[10]];
        MAP Y = save2y[save data[10]];
    }
    saveflagc = 0;
    Repeat (List.Length(save data))
    {
        saveflagc += 1;
        List.Add(flags, save data[saveflagc]);
        List.Add(Collecteditems, save data[saveflagc]);
    }
    Call replv(1);

    Call replv(3);
    If (AREA > 1)

    If (AREA > 2)

        List.ReplaceItem(Character HP, 3, 1);
    }
}

Define replv(string number or text) (warp=false)
{
    Repeat Until (((Character exp[number or text] - Exp Table[(Character levels[number or text] - 1)]) / (Exp Table[Character levels[number or text]] - Exp Table[(Character levels[number or text] - 1)])) < 0.9999)
    {
        List.ReplaceItem(Character levels, number or text, (Character levels[number or text] + 1));
        List.ReplaceItem(Character Stats AAADDDSSS, number or text, character stat tables[(Character levels[number or text] + ((number or text - 1) * 35))]);
    }
}

Define Story line add(string .) wait input(string wait input) end of section(string end) (warp=false)
{
    F = 0;
    story despite battle = 1;
    message delay = -23;
    List.Add(A Message Log, .);
    Control.Wait((wait input / 1.5));
    Repeat Until ((Sensing.MouseDown() Or Sensing.KeyPressed(any)) Or (Not ((wait input == 1))))
    {
        message delay = -6.1;
    }
    If (wait input == 1)
    {
        Sound.Play(Bling1);
    }
    Repeat (wait input)
    {
        List.DeleteAll(A Message Log);
    }
    If (Not ((end == 1)))
    {
        List.Add(A Message Log, " ");
    }
    Control.Wait(0.04);
    story despite battle = 0;
    List.DeleteItem(A Message Log, List.Length(A Message Log));
}
