//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked()
{
    Forever
    {
        Looks.SwitchCostumeTo(costume1);
        Looks.SetSizeTo(600);
        Looks.SwitchCostumeTo(AREA);
        Motion.GoToXY((0 - MAP X), (0 - MAP Y));
        Looks.Show();
        If (Battleover? == 0)
        {
            Looks.Hide();
        }
        Looks.SetEffectTo(BRIGHTNESS, 0);
    }
}
