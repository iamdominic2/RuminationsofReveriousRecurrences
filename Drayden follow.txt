//----- Green flag events ---------------------------------------------------------------------------------------------

WhenGreenFlagClicked() //follow 5 frames behind protagonist
{
    Looks.SetSizeTo(25);
    Forever
    {
        Motion.PointInDirection(followd[(List.Length(followx) - 8)]);
        Motion.GoToXY((followx[(List.Length(followx) - 8)] - MAP X), (followy[(List.Length(followx) - 8)] - MAP Y));
        Call FOUT;
        Looks.Hide();
        If ((Battleover? == 1) And (AREA > 1))
        {
            Looks.Show();
        }
    }
}

//----- Custom blocks -------------------------------------------------------------------------------------------------

Define FOUT (warp=true) //method to shift out of wall
{
    Looks.SetSizeTo(1);
    F = 1;
    Motion.ChangeYBy(-16);
    If (Not (Sensing.TouchingObject(Map)))
    {
        F = 0.01;
    }
    If (F == 1)
    {
        Repeat Until (F == 0.01)
        {
            Motion.Move(F);
            If (Not (Sensing.TouchingObject(Map)))
            {
                F = 0;
            }
            Motion.Move((0 - F));
            F += 0.01;
            Motion.TurnRight(22.5);
        }
        Motion.PointInDirection(Motion.XPosition());
        F = -1;
        Looks.SetSizeTo(1);
    }
    Motion.ChangeYBy(16);
    Looks.SetSizeTo(25);
}
